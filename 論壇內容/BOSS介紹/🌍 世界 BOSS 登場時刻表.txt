  <script>
    // 內容載入後的動作
    function bossLoaded() {
      // 1. 變更背景
      document.body.style.backgroundImage = 'none';
      document.body.style.backgroundColor = '#ffffff';

      // 2. 隱藏載入動畫
      const loading = document.getElementById('loading');
      loading.style.opacity = '0';
      setTimeout(() => {
        loading.style.display = 'none';
      }, 800);
      
      // 3. 顯示表格
      document.getElementById('boss-frame').style.opacity = '1';
    }

    // 關鍵修正 4：防呆機制
    // 如果 8 秒後 iframe 還沒觸發 onload (可能是 Google 伺服器慢或網路不穩)，強行進入頁面
    setTimeout(() => {
      const frame = document.getElementById('boss-frame');
      if (frame.style.opacity !== '1') {
        bossLoaded();
      }
    }, 8000);
  </script>
