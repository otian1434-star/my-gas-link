    function classLoaded() {
      // 1. 解除滾動限制
      const html = document.documentElement;
      const body = document.body;
      
      html.classList.add('loaded-body');
      body.classList.add('loaded-body');
      
      // 2. 處理 Loading 動畫與層級
      const loading = document.getElementById('loading');
      loading.style.opacity = '0';
      loading.style.pointerEvents = 'none'; // 關鍵：讓觸摸事件穿透，不干擾捲動
      
      // 3. 顯示 Iframe
      const frame = document.getElementById('class-frame');
      frame.style.opacity = '1';

      // 4. 背景色切換
      body.style.backgroundImage = 'none';
      body.style.backgroundColor = '#ffffff';

      // 5. 延遲徹底移除 Loading 節點
      setTimeout(() => {
        loading.style.display = 'none';
      }, 800);
    }

    // 備用方案：如果 Iframe 載入太慢，5秒後強制開啟捲動功能
    setTimeout(() => {
      if (document.getElementById('loading').style.display !== 'none') {
        classLoaded();
      }
    }, 5000);
  </script>
